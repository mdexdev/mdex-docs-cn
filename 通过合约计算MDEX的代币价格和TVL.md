一、通过交易对计算Token价格和TVL，以交易对MDX/USDT（0x615e6285c5944540fd8bd921c9c8c56739fd1e13）为例：

```javascript
let { ethers } = require("ethers");
let BigNumber = require("bignumber.js");
let pair_abi = require('./abis/pair_abi.json');
let erc20_abi = require('./abis/erc20_abi.json')

let rpc_url	= "https://http-mainnet-node.huobichain.com";
let pair_mdx_usdt	= "0x615e6285c5944540fd8bd921c9c8c56739fd1e13";

// 创建交易对合约，获取交易对MDX和USDT的数量
let heco_provider  =   new ethers.providers.JsonRpcProvider(rpc_url);
let pair_sol = new ethers.Contract(pair_mdx_usdt, pair_abi, heco_provider);
let pair_reserves	=	await pair_sol.getReserves();

// 获得交易对代币token0（MDX）的精度
let token0        =   await pair_sol.token0();
let token0_sol    =   erc20Contract(token0, eth_provider);
let decimal_0     =   await token0_sol.decimals();

// 获得交易对代币token1（USDT）的精度
let token1    		=   await pair_sol.token1();
let token1_sol    =   erc20Contract(token1, eth_provider);
let decimal_1     =   await token1_sol.decimals();

// ***********获取MDX的价格***********
let mdx_balance	= new BigNumber(pair_reserves['_reserve0'].toString()).div(Mqth.pow(10, decimal_0));
let usdt_balance = new BigNumber(pair_reserves['_reserve1'].toString()).div(Mqth.pow(10, decimal_1));
let mdx_price	=	usdt_balance.div(mdx_balance).toNumber();

// ***********获取交易对的TVL方式一***********
let mdx_usdt_tvl	= usdt_balance.times(2).toNumber();	

// ***********获取交易对的TVL方式二***********
let mdx_usdt_tvl	= mdx_balance.times(mdx_price).plus(usdt_balance.times(1)).toNumber();
```
